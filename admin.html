/** * Classroom Economy Brain */ function processTransaction(studentID, amount, type, description) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sSheet = ss.getSheetByName("Students"); const data = sSheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] == studentID) { let currentBal = parseFloat(data[i][3]) || 0; let change = (type === "Sale" || type === "Pool") ? -Math.abs(amount) : Math.abs(amount); if (currentBal + change < 0) return { success: false, message: "Insufficient Funds" }; sSheet.getRange(i + 1, 4).setValue(currentBal + change); ss.getSheetByName("Ledger").appendRow([new Date(), studentID, type, description, change, currentBal + change]); return { success: true }; } } return { success: false, message: "Student not found" }; } function payClass(amount) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sSheet = ss.getSheetByName("Students"); const data = sSheet.getDataRange().getValues(); const bonus = parseFloat(amount) || 0; for (let i = 1; i < data.length; i++) { let currentBalance = parseFloat(data[i][3]) || 0; sSheet.getRange(i + 1, 4).setValue(currentBalance + bonus); ss.getSheetByName("Ledger").appendRow([new Date(), data[i][0], "Reward", "Classwide Bonus", bonus, currentBalance + bonus]); } return { success: true }; } function poolTransaction(studentID, amount, description) { return processTransaction(studentID, amount, "Pool", description); } function loginStudent(name, password) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const data = ss.getSheetByName("Students").getDataRange().getValues(); const student = data.find(r => r[1].toLowerCase() === name.toLowerCase() && r[2].toString().toLowerCase() === password.toString().toLowerCase()); if (student) { const ledger = ss.getSheetByName("Ledger").getDataRange().getValues(); const history = ledger.filter(r => r[1] === student[0]); return { success: true, student: { id: student[0], name: student[1], balance: student[3] }, history: history }; } return { success: false }; } function getAllStudents() { const data = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Students").getDataRange().getValues(); return data.length > 1 ? data.slice(1) : []; } function addStudent(name, password) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName("Students"); const studentID = "STU-" + Math.floor(1000 + Math.random() * 9000); sheet.appendRow([studentID, name, password, 0]); return { success: true, id: studentID }; } // --- CARD GENERATION --- function generateStudentCard(studentId) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const data = ss.getSheetByName("Students").getDataRange().getValues(); let studentData = data.find(r => r[0].toString() === studentId.toString()); if (!studentData) return "Error: Student not found"; const templateId = "1ZBbs_vtIMVuC5OCkGgVfmuub5jToX8nYEe-Z5Kpm2Fc"; const presentation = DriveApp.getFileById(templateId).makeCopy("Card_" + studentData[1]); const slideApp = SlidesApp.openById(presentation.getId()); const slide = slideApp.getSlides()[0]; slide.replaceAllText('{{NAME}}', studentData[1]); const qrUrl = "https://quickchart.io/qr?text=" + studentData[0] + "&size=200"; slide.insertImage(qrUrl, 450, 100, 150, 150); slideApp.saveAndClose(); return presentation.getUrl(); } // --- WEB APP HANDLER --- function doPost(e) { var p = JSON.parse(e.postData.contents); var res; switch(p.action) { case "getItems": res = getItems(); break; case "getGoals": res = getGoals(); break; case "processTransaction": res = processTransaction(p.studentID, p.amount, p.type, p.description); break; case "payClass": res = payClass(p.amount); break; case "poolTransaction": res = poolTransaction(p.studentID, p.amount, p.description); break; case "getAllStudents": res = getAllStudents(); break; case "addStudent": res = addStudent(p.name, p.password); break; case "loginStudent": res = loginStudent(p.name, p.password); break; case "generateStudentCard": res = generateStudentCard(p.studentId); break; } return ContentService.createTextOutput(JSON.stringify(res)).setMimeType(ContentService.MimeType.JSON); } function getItems() { return SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Inventory").getDataRange().getValues().slice(1); } function getGoals() { return SpreadsheetApp.getActiveSpreadsheet().getSheetByName("IncomeGoals").getDataRange().getValues().slice(1); } function doGet() { return ContentService.createTextOutput("System Active"); }
